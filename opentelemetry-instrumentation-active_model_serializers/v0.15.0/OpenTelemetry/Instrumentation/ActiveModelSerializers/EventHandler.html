def handle(start_timestamp, end_timestamp, payload)
  tracer.start_span(span_name(payload),
                    start_timestamp: start_timestamp,
                    attributes: build_attributes(payload),
                    kind: :internal)
        .finish(end_timestamp: end_timestamp)
end
