def initialize(req, options)
  attributes = {
    'net.peer.name' => req.uri.host,
    'net.peer.port' => req.uri.port
  }.merge!(OpenTelemetry::Common::HTTP::ClientContext.attributes)

  tracer.in_span('HTTP CONNECT', attributes: attributes) do
    super
  end
end
