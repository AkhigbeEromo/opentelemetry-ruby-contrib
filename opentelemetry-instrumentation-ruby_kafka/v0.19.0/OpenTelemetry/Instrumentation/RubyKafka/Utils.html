def extract_message_key(key)
  # skip encode if already valid utf8
  return key if key.nil? || (key.encoding == Encoding::UTF_8 && key.valid_encoding?)

  key.encode(Encoding::UTF_8)
rescue Encoding::UndefinedConversionError
  nil
end
